all_data = FileAttachment("dashboard_data.csv").csv({typed: true})

// 2. Explicitly require D3
d3 = require("d3@7")

// 3. Centralized Filtering (Numeric comparison for stability)
// Scenario A logic
data_A_raw = all_data.filter(d => 
  Number(d.wind_threshold) === Number(wind_A) && 
  Number(d.wave_threshold) === Number(wave_A)
)

// Scenario B logic
data_B_raw = all_data.filter(d => 
  Number(d.wind_threshold) === Number(wind_B) && 
  Number(d.wave_threshold) === Number(wave_B)
)

// 4. Shared Rollups (Used by both plots and tables)
data_A_daily = Array.from(d3.rollup(data_A_raw, v => v[0], d => d.contract_day_index).values())
data_B_daily = Array.from(d3.rollup(data_B_raw, v => v[0], d => d.contract_day_index).values())

data_A_pricing = Array.from(d3.rollup(data_A_raw, v => v[0], d => d.strike_percent).values())
data_B_pricing = Array.from(d3.rollup(data_B_raw, v => v[0], d => d.strike_percent).values())
// No redundant filtering here, just the plot
Plot.plot({
  y: {grid: true, domain: [0, 1]},
  marks: [

  ]
})

})
